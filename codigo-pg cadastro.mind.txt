Mostra um formulário de cadastro de evento.Quando clica no botão "Criar evento", o evento é salvo temporariamente.Redireciona para a página de aprovação e exibe os eventos cadastrados.Código único em Python Código:from flask import Flask, request, redirect, render_template_stringapp = Flask(__name__)eventos = []TEMPLATE = '''<!DOCTYPE html><html><head>    <title>{{ titulo }}</title></head><body>    {% if pagina == 'cadastro' %}        <h2>Cadastro de Evento</h2>        <form method="POST">            <input type="text" name="nome" placeholder="Nome do evento" required><br>            <input type="date" name="data" required><br>            <label><input type="radio" name="entrada" value="Gratuita" required> Gratuita</label>            <label><input type="radio" name="entrada" value="Paga"> Paga</label><br>            <input type="number" name="valor" placeholder="Valor (R$)"><br>            <textarea name="descricao" placeholder="Descrição" required></textarea><br>            <button type="submit">Criar evento</button>        </form>    {% elif pagina == 'aprovacao' %}        <h2>Eventos Pendentes de Aprovação</h2>        {% if eventos %}            {% for evento in eventos %}                <div style="border: 1px solid #ccc; padding: 10px; margin: 10px;">                    <strong>Nome:</strong> {{ evento.nome }}<br>                    <strong>Data:</strong> {{ evento.data }}<br>                    <strong>Tipo:</strong> {{ evento.entrada }}<br>                    <strong>Valor:</strong> R$ {{ evento.valor }}<br>                    <strong>Descrição:</strong> {{ evento.descricao }}<br>                    <button>Aprovar</button>                    <button>Rejeitar</button>                </div>            {% endfor %}        {% else %}            <p>Nenhum evento cadastrado.</p>        {% endif %}    {% endif %}</body></html>'''@app.route('/', methods=['GET', 'POST'])def cadastro():    if request.method == 'POST':        evento = {            'nome': request.form['nome'],            'data': request.form['data'],            'entrada': request.form['entrada'],            'valor': request.form['valor'],            'descricao': request.form['descricao']        }        eventos.append(evento)        return redirect('/aprovacao')    return render_template_string(TEMPLATE, titulo='Cadastro de Evento', pagina='cadastro')@app.route('/aprovacao')def aprovacao():    return render_template_string(TEMPLATE, titulo='Aprovação de Evento', pagina='aprovacao', eventos=eventos)if __name__ == '__main__':    app.run(debug=True)
